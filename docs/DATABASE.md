# IIIS Database Schema Specification

This document defines the 16 database tables for the IIIS v4.6 system (PostgreSQL + TimescaleDB) as defined in the frozen specification.

---

## 16 Core Tables

### TABLE 1: market_data (TimescaleDB Hypertable)
Used for storing historical and real-time market data ticks/candles across different timeframes.
```sql
CREATE TABLE market_data (
    time TIMESTAMPTZ NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    open DECIMAL(12,4),
    high DECIMAL(12,4),
    low DECIMAL(12,4),
    close DECIMAL(12,4),
    volume BIGINT,
    vwap DECIMAL(12,4),
    timeframe VARCHAR(10) NOT NULL
);

SELECT create_hypertable('market_data', 'time', chunk_time_interval => INTERVAL '1 day');
```

### TABLE 2: signals
Stores all trade signals generated by the system after passing upstream checks and validation.
```sql
CREATE TABLE signals (
    signal_id VARCHAR(30) PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    direction VARCHAR(10) NOT NULL CHECK (direction IN ('LONG', 'SHORT')),
    score DECIMAL(5,2) NOT NULL CHECK (score > 85.0),
    confidence VARCHAR(10) NOT NULL CHECK (confidence IN ('HIGH', 'MEDIUM', 'LOW')),
    regime VARCHAR(20) NOT NULL,
    entry_low DECIMAL(12,4) NOT NULL,
    entry_high DECIMAL(12,4) NOT NULL,
    stop_loss DECIMAL(12,4) NOT NULL,
    target_1 DECIMAL(12,4) NOT NULL,
    target_2 DECIMAL(12,4) NOT NULL,
    quantity INTEGER NOT NULL,
    risk_amount DECIMAL(12,2) NOT NULL,
    geie_direction VARCHAR(20),
    geie_confidence VARCHAR(10),
    arc_decision VARCHAR(20),
    historical_wr DECIMAL(5,2),
    historical_sample INTEGER,
    risk_grade VARCHAR(5) NOT NULL,
    status VARCHAR(20) DEFAULT 'ACTIVE' CHECK (status IN ('ACTIVE', 'EXPIRED', 'HIT_SL', 'HIT_T1', 'HIT_T2', 'CANCELLED')),
    valid_until TIMESTAMPTZ NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);
```

### TABLE 3: active_alerts
Tracks currently active alerts in the market and expiration thresholds.
```sql
CREATE TABLE active_alerts (
    alert_id SERIAL PRIMARY KEY,
    signal_id VARCHAR(30) REFERENCES signals(signal_id),
    symbol VARCHAR(20) NOT NULL,
    direction VARCHAR(10) NOT NULL,
    sector VARCHAR(50) NOT NULL,
    triggered_at TIMESTAMPTZ NOT NULL,
    expires_at TIMESTAMPTZ NOT NULL
);
```

### TABLE 4: risk_state
Tracks risk parameters, daily risk usage, and system hard stops.
```sql
CREATE TABLE risk_state (
    session_date DATE UNIQUE NOT NULL,
    daily_risk_used DECIMAL(5,2) DEFAULT 0,
    consecutive_losses INTEGER DEFAULT 0,
    hard_stop_active BOOLEAN DEFAULT FALSE,
    total_signals INTEGER DEFAULT 0,
    signals_hit_sl INTEGER DEFAULT 0,
    signals_hit_t1 INTEGER DEFAULT 0,
    signals_hit_t2 INTEGER DEFAULT 0,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

### TABLE 5: regime_history
Records market regime calculations and history.
```sql
CREATE TABLE regime_history (
    id SERIAL PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    regime VARCHAR(20) NOT NULL,
    regime_score DECIMAL(5,2) NOT NULL,
    adx DECIMAL(8,4),
    nifty_price DECIMAL(12,4),
    ad_ratio DECIMAL(5,2),
    notes TEXT
);
```

### TABLE 6: audit_log
Immutable log of system-level actions and decisions for compliance and auditing.
> [!IMPORTANT]
> **RULE:** Never delete. Never update. Append only forever.
```sql
CREATE TABLE audit_log (
    id BIGSERIAL PRIMARY KEY,
    timestamp TIMESTAMPTZ DEFAULT NOW(),
    component VARCHAR(50) NOT NULL,
    action VARCHAR(100) NOT NULL,
    result VARCHAR(50) NOT NULL,
    reason TEXT,
    metadata JSONB
);

CREATE INDEX idx_audit_timestamp ON audit_log(timestamp);
```

### TABLE 7: system_health
Used for system health check status monitoring.
```sql
CREATE TABLE system_health (
    id SERIAL PRIMARY KEY,
    checked_at TIMESTAMPTZ NOT NULL,
    component VARCHAR(50) NOT NULL,
    status VARCHAR(20) NOT NULL,
    response_time_ms INTEGER,
    ghost_mode_active BOOLEAN DEFAULT FALSE,
    last_error TEXT
);
```

### TABLE 8: geie_events
Stores global macro news events processed by the GEIE engine.
```sql
CREATE TABLE geie_events (
    event_id VARCHAR(50) PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    event_name VARCHAR(100),
    impact_direction VARCHAR(20),
    confidence VARCHAR(10),
    urgency VARCHAR(20),
    beneficiaries JSONB,
    losers JSONB,
    neutral JSONB,
    raw_output JSONB
);
```

### TABLE 9: geie_master_map
Holds master associations between stock symbols and positive/negative/neutral GEIE triggers.
```sql
CREATE TABLE geie_master_map (
    symbol VARCHAR(20) PRIMARY KEY,
    positive_triggers TEXT[],
    negative_triggers TEXT[],
    neutral_triggers TEXT[],
    last_updated TIMESTAMPTZ DEFAULT NOW()
);
```

### TABLE 10: options_data
Contains option chain data points (OI, volume, IV, etc.) for options intelligence.
```sql
CREATE TABLE options_data (
    time TIMESTAMPTZ NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    strike DECIMAL(12,4) NOT NULL,
    expiry DATE NOT NULL,
    option_type VARCHAR(5) NOT NULL CHECK (option_type IN ('CE', 'PE')),
    oi BIGINT,
    oi_change BIGINT,
    volume BIGINT,
    iv DECIMAL(8,4),
    ltp DECIMAL(12,4),
    PRIMARY KEY (time, symbol, strike, expiry, option_type)
);
```

### TABLE 11: earnings_calendar
Tracks stock earnings calendar dates and times.
```sql
CREATE TABLE earnings_calendar (
    symbol VARCHAR(20) NOT NULL,
    earnings_date DATE NOT NULL,
    earnings_time TIME,
    PRIMARY KEY (symbol, earnings_date)
);
```

### TABLE 12: event_calendar
Tracks scheduled economic events and macroeconomic indicators.
```sql
CREATE TABLE event_calendar (
    event_name VARCHAR(100) NOT NULL,
    event_date DATE NOT NULL,
    event_time TIME,
    impact_level VARCHAR(20) NOT NULL,
    description TEXT,
    PRIMARY KEY (event_name, event_date)
);
```

### TABLE 13: fii_dii_tracker (v4.6)
Tracks daily institutional FII/DII activities and combined sentiment.
```sql
CREATE TABLE fii_dii_tracker (
    date DATE PRIMARY KEY,
    fii_action VARCHAR(10) NOT NULL CHECK (fii_action IN ('BUYER', 'SELLER')),
    fii_amount_crores DECIMAL(12,2),
    dii_action VARCHAR(10) NOT NULL CHECK (dii_action IN ('BUYER', 'SELLER')),
    dii_amount_crores DECIMAL(12,2),
    combined_bias VARCHAR(20) NOT NULL CHECK (combined_bias IN ('BULLISH', 'BEARISH', 'NEUTRAL')),
    consecutive_fii_buy_days INTEGER DEFAULT 0,
    consecutive_fii_sell_days INTEGER DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW()
);
```

### TABLE 14: bulk_deal_tracker (v4.6)
Monitors large institutional block and bulk deals on the exchange.
```sql
CREATE TABLE bulk_deal_tracker (
    id SERIAL PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    deal_type VARCHAR(10) NOT NULL CHECK (deal_type IN ('BUY', 'SELL')),
    quantity BIGINT,
    price DECIMAL(12,4),
    value_crores DECIMAL(12,2),
    client_name VARCHAR(200),
    deal_category VARCHAR(10) NOT NULL CHECK (deal_category IN ('BULK', 'BLOCK')),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_bulk_symbol ON bulk_deal_tracker(symbol);
CREATE INDEX idx_bulk_timestamp ON bulk_deal_tracker(timestamp);
```

### TABLE 15: options_intelligence (v4.6)
Keeps summarized options intelligence metrics for each session date and symbol.
```sql
CREATE TABLE options_intelligence (
    date DATE NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    max_pain_level DECIMAL(12,4),
    highest_put_oi_strike DECIMAL(12,4),
    highest_call_oi_strike DECIMAL(12,4),
    unusual_activity_detected BOOLEAN DEFAULT FALSE,
    unusual_strike DECIMAL(12,4),
    unusual_type VARCHAR(10) CHECK (unusual_type IN ('CALL', 'PUT')),
    unusual_oi_change BIGINT,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    PRIMARY KEY (date, symbol)
);
```

### TABLE 16: order_block_memory (v4.6)
Stores identified smart money Order Blocks (OB) and historical performance/test data.
```sql
CREATE TABLE order_block_memory (
    id SERIAL PRIMARY KEY,
    symbol VARCHAR(20) NOT NULL,
    timeframe VARCHAR(10) NOT NULL CHECK (timeframe IN ('5m', '15m', 'Daily')),
    ob_type VARCHAR(10) NOT NULL CHECK (ob_type IN ('BULLISH', 'BEARISH')),
    ob_high DECIMAL(12,4) NOT NULL,
    ob_low DECIMAL(12,4) NOT NULL,
    ob_midpoint DECIMAL(12,4) NOT NULL,
    first_detected TIMESTAMPTZ NOT NULL,
    last_tested TIMESTAMPTZ,
    test_count INTEGER DEFAULT 0,
    held_count INTEGER DEFAULT 0,
    broken BOOLEAN DEFAULT FALSE,
    broken_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_ob_symbol ON order_block_memory(symbol);
CREATE INDEX idx_ob_broken ON order_block_memory(broken) WHERE broken = FALSE;
```
